### Task 2: CI matrix expansion

**Files:**
- Modify: `.github/workflows/ci.yml`

**Step 1: Expand the CI workflow**

Replace the entire `ci.yml` with an expanded matrix:

```yaml
name: CI

on:
  push:
    branches: [main, 'release/*']
  pull_request:
    branches: [main]

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

jobs:
  quality:
    name: Lint & Type-check (${{ matrix.os }})
    runs-on: ${{ matrix.os }}
    strategy:
      fail-fast: false
      matrix:
        os: [ubuntu-latest, windows-latest, macos-latest]
    steps:
      - uses: actions/checkout@v6
      - uses: actions/setup-node@v6
        with: { node-version: '25', cache: 'npm' }
      - run: npm ci
      - run: npm run lint
      - run: npm run typecheck
      - run: npx tsgo --noEmit --skipLibCheck
      - run: npm run build
      - run: npx publint

  test-unit:
    name: Unit Tests (${{ matrix.os }}, Node ${{ matrix.node }})
    runs-on: ${{ matrix.os }}
    needs: quality
    strategy:
      fail-fast: false
      matrix:
        os: [ubuntu-latest, windows-latest, macos-latest]
        node: ['25']
    steps:
      - uses: actions/checkout@v6
      - uses: actions/setup-node@v6
        with: { node-version: ${{ matrix.node }}, cache: 'npm' }
      - run: npm ci
      - run: npm run build
      - run: npm run test -- --project unit --coverage
      - uses: codecov/codecov-action@v5
        if: github.event_name == 'pull_request' && matrix.os == 'ubuntu-latest'

  test-browser:
    name: Browser Tests — ${{ matrix.browser }}
    runs-on: ubuntu-latest
    needs: quality
    strategy:
      fail-fast: false
      matrix:
        browser: [chromium, firefox, webkit]
    steps:
      - uses: actions/checkout@v6
      - uses: actions/setup-node@v6
        with: { node-version: '25', cache: 'npm' }
      - run: npm ci
      - run: npm run build
      - run: npx playwright install --with-deps ${{ matrix.browser }}
      - run: npm run test -- --project browser:${{ matrix.browser }}

  size:
    name: Bundle Size
    runs-on: ubuntu-latest
    needs: quality
    steps:
      - uses: actions/checkout@v6
      - uses: actions/setup-node@v6
        with: { node-version: '25', cache: 'npm' }
      - run: npm ci
      - run: npm run build
      - run: npm run size
      - uses: andresz1/size-limit-action@v1
        if: github.event_name == 'pull_request'
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}

  audit:
    name: Security Audit
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v6
      - uses: actions/setup-node@v6
        with: { node-version: '25', cache: 'npm' }
      - run: npm ci
      - run: npm audit --omit=dev

  socket:
    name: Socket Security
    runs-on: ubuntu-latest
    if: github.event_name == 'pull_request'
    permissions:
      contents: read
      pull-requests: write
    steps:
      - uses: actions/checkout@v6
      - uses: socket-security/socket-security-action@v1
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
```

**Step 2: Verify YAML syntax**

**Files:**
- Modify: `.github/workflows/release.yml`

**Step 1: Add npm pack verification and audit to release workflow**

```yaml
name: Release

on:
  push:
    tags: ['v*']

concurrency:
  group: release-${{ github.ref }}
  cancel-in-progress: false

jobs:
  release:
    name: Release to npm
    runs-on: ubuntu-latest
    environment: release
    permissions:
      contents: write
      id-token: write
    steps:
      - uses: actions/checkout@v6
        with: { fetch-depth: 0 }
      - uses: actions/setup-node@v6
        with:
          node-version: '25'
          cache: 'npm'
          registry-url: 'https://registry.npmjs.org'
      - run: npm ci
      - run: npm audit --omit=dev
      - run: npm run build
      - run: npm run test -- --project unit
      - run: npm run size
      - run: npx publint
      - run: npm pack --dry-run
      - run: npm publish --provenance --access public
        env:
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}
```

**Step 2: Commit**

```bash
git add .github/workflows/release.yml
git commit -m "ci: add audit, publint, and pack verification to release workflow"
```
